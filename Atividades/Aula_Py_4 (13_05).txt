Exercício 4


num1 = int(input("Digite o primeiro número: "))
num2 = int(input("Digite o segundo número: "))


pares = 0
impares = 0


for i in range(num1, num2 + 1):
    if i % 2 == 0:
        pares += 1
    else:
        impares += 1


print("Existem", pares, "números pares e", impares, "números ímpares entre", num1, "e", num2, ".")


Exercício 5


N = int(input("Digite um número inteiro positivo N: "))


if N <= 0:
    print("Por favor, digite um número inteiro positivo.")
else:
    soma = 0
    for i in range(1, N + 1):
        soma += i


    print("A soma dos primeiros", N, "números naturais é:", soma)


Exercício 6


num_notas = int(input("Digite a quantidade de notas que deseja inserir: "))


soma_notas = 0


for i in range(num_notas):
    nota = float(input("Digite a nota: ".format(i+1)))
    soma_notas += nota


media = soma_notas / num_notas


print("A média das", num_notas, "notas inseridas é:", media)
